Option Explicit

' --- OBSŁUGA KLIKNIĘCIA PRZYCISKÓW KANAPEK ---
Private Sub cmdKurczak_Click()
    GenerateAndPrintLabel "KANAPKA Z KURCZAKIEM"
End Sub

Private Sub cmdWegetarianska_Click()
    GenerateAndPrintLabel "KANAPKA WEGETARIAŃSKA"
End Sub

Private Sub cmdLosos_Click()
    GenerateAndPrintLabel "KANAPKA Z ŁOSOSIEM"
End Sub

' --- FUNKCJA GENERUJĄCA ETYKIETĘ ---
Sub GenerateAndPrintLabel(ByVal sandwichName As String)
    Dim wsLabels As Worksheet
    Dim printRange As Range

    ' Usunięcie starego arkusza etykiet, jeśli istnieje
    On Error Resume Next
    Application.DisplayAlerts = False
    Sheets("Etykiety").Delete
    Application.DisplayAlerts = True
    On Error GoTo 0

    ' Tworzenie nowego arkusza etykiet
    Set wsLabels = ThisWorkbook.Sheets.Add
    wsLabels.Name = "Etykiety"
    wsLabels.Activate

    ' Konfiguracja szerokości komórki (62 mm)
    wsLabels.Columns("A").ColumnWidth = 30.14

    ' Ustalenie treści etykiety
    Dim labelContent As String
    Select Case sandwichName
        Case "KANAPKA Z KURCZAKIEM"
            labelContent = GenerateChickenSandwichLabel()
        Case "KANAPKA WEGETARIAŃSKA"
            labelContent = "MASA NETTO: 180 g" & vbNewLine & _
                           "KANAPKA WEGETARIAŃSKA" & vbNewLine & _
                           "Składniki: ..."
        Case "KANAPKA Z ŁOSOSIEM"
            labelContent = "MASA NETTO: 200 g" & vbNewLine & _
                           "KANAPKA Z ŁOSOSIEM" & vbNewLine & _
                           "Składniki: ..."
    End Select

    ' Wpisanie etykiety do jednej komórki z zawijaniem tekstu
    With wsLabels.Range("A1")
        .Value = labelContent
        .WrapText = True  ' Zawijanie tekstu w komórce
        .Font.Size = 6
        .Font.Bold = False
        .HorizontalAlignment = xlLeft  ' Wyrównanie do lewej
        .VerticalAlignment = xlTop     ' Tekst od góry
        .Borders.LineStyle = xlNone
    End With

    ' ?? AUTOMATYCZNE DOPASOWANIE WYSOKOŚCI KOMÓRKI DO TEKSTU
    wsLabels.Rows("1:1").AutoFit

    ' Zakres do druku – tylko A1
    Set printRange = wsLabels.Range("A1")

    With wsLabels.PageSetup
        .PrintArea = printRange.Address
        .Orientation = xlPortrait  ' Druk pionowo
        .Zoom = False
        .FitToPagesWide = 1
        .FitToPagesTall = False
        .LeftMargin = Application.InchesToPoints(0)
        .RightMargin = Application.InchesToPoints(0)
        .TopMargin = Application.InchesToPoints(0)
        .BottomMargin = Application.InchesToPoints(0.5)
    End With

    ' --- WYBÓR DRUKARKI PRZED DRUKIEM ---
    Application.Dialogs(xlDialogPrint).Show

    ' Usunięcie arkusza po wydruku
    Application.DisplayAlerts = False
    wsLabels.Delete
    Application.DisplayAlerts = True
End Sub
